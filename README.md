# myTmuxSetup

## tmux setup
$ mv tmuxconfig ~/.tmux.conf
version >= 2.0
use Bind-key + :
and type source ~/.tmux.conf

## vim setup
$ mv vimrc ~/.vimrc

### for color scheme
$ git clone https://github.com/tomasr/molokai.git
$ mkdir ~/.vim/
$ mv colors ~/.vim/

## bash setup
$ mv bashrc ~/.bashrc
$ source ~/.bashrc
